const EVALUATOR_PROMPT = (testCase, actualOutput) => `You are an expert AI product tester evaluating a prompt output. 

TEST CASE: ${testCase.name}
INPUT GIVEN TO PROMPT: ${testCase.input}
EXPECTED BEHAVIOUR: ${testCase.expectedBehaviour}

ACTUAL OUTPUT FROM PROMPT:
${actualOutput}

Evaluate this output against the expected behaviour and the following quality criteria from the prompt spec:
1. Does it produce 3–5 pedagogical personas with ALL required attributes (Name/Role, Background, Bloom's Level, Motivation, Friction Points, Success Criterion)?
2. Does chain-of-thought reasoning precede each persona?
3. Does it cite evidence from the provided research?
4. Do personas differ on ≥2 dimensions?
5. Are validation checks (coverage, differentiation, aspirational, Bloom's distribution) present and honest?
6. Does it comply with R12 Input Fidelity (flag thin research, don't fabricate)?
7. Does it correctly handle the task type (pedagogical vs marketing personas)?

Return ONLY a JSON object with these exact keys:
{
  "result": "pass" | "partial" | "fail",
  "actual_behaviour_summary": "2-3 sentence factual description of what the output actually did",
  "prompt_edit_required": "Specific edit needed, or 'none' if passes"
}`;

const callClaude = async (systemPrompt, userMessage, isEvaluator = false) => {
  const response = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: systemPrompt,
      messages: [{ role: "user", content: userMessage }],
    }),
  });
  const data = await response.json();
  if (!response.ok) throw new Error(data.error?.message || "API error");
  return data.content.map(b => b.text || "").join("\n");
};
